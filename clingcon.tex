  \begin{itemize}
  \item \structure{Idea} \ extend \clingo\ with constraints over integers
  \item \structure{Features} \

    \end{itemize}
  \item \structure{References} \ \cite{bakaossc16a}
  \item \structure{Applications} \ incorporate quantities, like resources and/or time
  \end{itemize}
